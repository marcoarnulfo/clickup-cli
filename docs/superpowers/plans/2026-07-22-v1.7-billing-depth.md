# v1.7 — Billing depth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the report engine into a billing engine: billable split, per-member/(list,member) rates, rounding, multi-currency subtotals, budgets, timezone correctness, weekly reports, tag grouping, and non-fiscal HTML/CSV-invoice export.

**Architecture:** All pricing/rounding/budget/currency math stays PURE in `internal/report` (which may import `internal/duration`, also pure) — no I/O, no `time.Now`. Impure wiring (config → pricing, location loading) lives in `internal/service`; TUI and `internal/cli` consume it. `report.Build` is rewritten once, early, behind a new `Pricing` input; later tasks graft on top.

**Tech Stack:** Go 1.26, bubbletea/lipgloss/bubbles, cobra, `gopkg.in/yaml.v3`, stdlib `html/template`/`encoding/csv`.

**Spec:** `docs/superpowers/specs/2026-07-22-v1.7-billing-depth-design.md` (read it; this plan implements it).

## Global Constraints

- **Purity:** `internal/report` and `internal/duration` never do I/O, never call `time.Now`, never import `config`/`clickup`/`tui`. `report` MAY import `duration`. Any violation is a defect.
- **Money ledger rule (single, everywhere):** the monetary amount is rounded to 2 decimals at the smallest rate-homogeneous **billing unit** (PerEntry → each billable entry; PerDay → each `(day-in-loc, listID, userID)` group). Every aggregate (bucket, currency subtotal, total) is a **sum of already-rounded unit amounts**. CSV-invoice rows ARE the billing units, so `sum(CSV amounts) == CurrencySubtotals == HTML totals`. A cross-export equality test enforces this.
- **Timezone two-track:** ranges take `loc *time.Location`; `loc == nil` ⇒ UTC. **Headless `clup report` defaults to UTC** (never changes silently) unless `timezone` in config or `--tz` is given. **TUI defaults to `time.Local`** (or config `timezone`). The JSON output declares the timezone used and carries a `schema_version`.
- **Billable default:** decode ClickUp `billable` as `*bool`; absent ⇒ `Billable = true` (preserves today's "bill everything").
- **No cross-currency totals:** `CurrencySubtotals` is the source of truth; `Report.TotalAmount` is populated ONLY when the report is single-currency, else `0`.
- **HTML export:** self-contained, CSS inline, **no remote assets/fonts** (golden test forbids them).
- **Process:** Conventional Commits, **NEVER `Co-Authored-By`**. In-repo text in English (this plan's prose is Italian per repo convention; code/tests/UI/commits English). Before every commit: `gofmt -l .` (empty), `go vet ./...`, `go build ./...`, `go test ./... -race` (green).
- **Demo parity:** every user-visible feature works under `CLICKUP_DEMO=1` with zero I/O.

```go
// midnightIn returns the start-of-day of t's calendar date in loc.
func midnightIn(t time.Time, loc *time.Location) time.Time {
	if loc == nil {
		loc = time.UTC
	}
	t = t.In(loc)
	return time.Date(t.Year(), t.Month(), t.Day(), 0, 0, 0, 0, loc)
}

// WeekRange returns [start, end) for the ISO-8601 week (Mon..Sun) in loc.
func WeekRange(isoYear, isoWeek int, loc *time.Location) (start, end time.Time) {
	if loc == nil {
		loc = time.UTC
	}
	// Jan 4 is always in ISO week 1; walk back to that week's Monday, then add weeks.
	jan4 := time.Date(isoYear, 1, 4, 0, 0, 0, 0, loc)
	wd := int(jan4.Weekday())
	if wd == 0 {
		wd = 7
	}
	week1Mon := jan4.AddDate(0, 0, -(wd - 1))
	start = week1Mon.AddDate(0, 0, (isoWeek-1)*7)
	end = start.AddDate(0, 0, 7)
	return start, end
}
```

```go
// FormatHours renders a duration as decimal hours with 2 decimals, e.g. "1.50h".
func FormatHours(d time.Duration) string {
	return strconv.FormatFloat(d.Hours(), 'f', 2, 64) + "h"
}

type RoundMode int

const (
	RoundNearest RoundMode = iota
	RoundUp
)

// Round rounds d to the nearest multiple of increment (RoundNearest) or up to it
// (RoundUp). increment <= 0 returns d unchanged. Negative d is returned unchanged.
func Round(d, increment time.Duration, mode RoundMode) time.Duration {
	if increment <= 0 || d <= 0 {
		return d
	}
	q := d / increment
	r := d % increment
	switch mode {
	case RoundUp:
		if r > 0 {
			q++
		}
	default: // RoundNearest
		if r*2 >= increment {
			q++
		}
	}
	return q * increment
}
```

```go
func Build(entries []TimeEntry, groupBy string, p Pricing, start, end time.Time, loc *time.Location) Report {
	if loc == nil {
		loc = time.UTC
	}
	r := Report{Start: start, End: end, GroupBy: groupBy, DefaultCurrency: p.DefaultCurrency, Timezone: loc.String()}

	// --- billing units: rate-homogeneous groups over BILLABLE entries ---
	// unitKey: PerDay -> "day|list|user"; PerEntry -> entry ID.
	type unit struct {
		listID string
		userID int
		raw    time.Duration
		ents   []int // indexes into entries, for finer-than-unit grouping (task/tag)
	}
	units := map[string]*unit{}
	var unitOrder []string
	for i, e := range entries {
		if !e.Billable {
			continue
		}
		var k string
		if p.Rounding.Scope == PerDay && p.Rounding.Increment > 0 {
			day := e.Start.In(loc).Format("2006-01-02")
			k = day + "|" + e.ListID + "|" + strconv.Itoa(e.UserID)
		} else {
			k = "e|" + e.ID
		}
		u := units[k]
		if u == nil {
			u = &unit{listID: e.ListID, userID: e.UserID}
			units[k] = u
			unitOrder = append(unitOrder, k)
		}
		u.raw += e.Duration
		u.ents = append(u.ents, i)
	}

	// Per-currency accumulators (authoritative totals).
	curH := map[string]float64{}   // raw billable hours by currency
	curBilled := map[string]float64{}
	curAmt := map[string]float64{}
	// Per-bucket accumulators.
	type bacc struct {
		label                        string
		hours, billableHours, billed float64
		amt                          map[string]float64
	}
	buckets := map[string]*bacc{}
	var bOrder []string
	getBucket := func(key, label string) *bacc {
		b := buckets[key]
		if b == nil {
			b = &bacc{label: label, amt: map[string]float64{}}
			buckets[key] = b
			bOrder = append(bOrder, key)
		}
		return b
	}

	// 1) All-entries hours (raw), billable/non split, and bucket raw hours.
	//    Group-by-tag duplicates an entry into each of its tags.
	for _, e := range entries {
		h := e.Duration.Hours()
		for _, gk := range groupKeys(e, groupBy, loc) {
			b := getBucket(gk.key, gk.label)
			b.hours += h
			if e.Billable {
				b.billableHours += h
			}
		}
	}

	// 2) Billed hours + amounts from units (ledger rule).
	for _, k := range unitOrder {
		u := units[k]
		billed := duration.Round(u.raw, p.Rounding.Increment, p.Rounding.Mode)
		rate := p.Rates.For(u.listID, u.userID)
		cur := p.currencyFor(u.listID)
		amt := round2(billed.Hours() * rate)
		curH[cur] += u.raw.Hours()
		curBilled[cur] += billed.Hours()
		curAmt[cur] += amt

		// Attribute billed hours + amount to buckets. If the grouping is not
		// finer than the unit, all the unit's entries share one bucket key;
		// otherwise split proportionally by raw hours (PerDay + task/tag only).
		if len(u.ents) == 1 || sameBucketForAll(u.ents, entries, groupBy, loc) {
			e := entries[u.ents[0]]
			for _, gk := range groupKeys(e, groupBy, loc) {
				b := getBucket(gk.key, gk.label)
				b.billed += billed.Hours()
				b.amt[cur] += amt
			}
		} else {
			for _, idx := range u.ents {
				e := entries[idx]
				share := 0.0
				if u.raw > 0 {
					share = e.Duration.Hours() / u.raw.Hours()
				}
				for _, gk := range groupKeys(e, groupBy, loc) {
					b := getBucket(gk.key, gk.label)
					b.billed += billed.Hours() * share
					b.amt[cur] += amt * share
				}
			}
		}
	}

	// --- assemble Report ---
	for _, cur := range sortedKeys(curAmt, curH, curBilled) {
		r.CurrencySubtotals = append(r.CurrencySubtotals, CurrencySubtotal{
			Currency: cur, Hours: round2(curH[cur]), BilledHours: round2(curBilled[cur]), Amount: round2(curAmt[cur]),
		})
	}
	if len(r.CurrencySubtotals) == 1 {
		r.TotalAmount = r.CurrencySubtotals[0].Amount
	}
	for _, k := range bOrder {
		b := buckets[k]
		r.Buckets = append(r.Buckets, Bucket{
			Label: b.label, Key: k, Hours: round2(b.hours), BillableHours: round2(b.billableHours),
			BilledHours: round2(b.billed), Amounts: currencyAmounts(b.amt),
		})
	}
	// totals
	for _, e := range entries {
		r.TotalHours += e.Duration.Hours()
		if !e.Billable {
			r.NonBillableHours += e.Duration.Hours()
		} else {
			r.BillableHours += e.Duration.Hours()
		}
	}
	r.TotalHours = round2(r.TotalHours)
	r.BillableHours = round2(r.BillableHours)
	r.NonBillableHours = round2(r.NonBillableHours)
	for _, s := range r.CurrencySubtotals {
		r.BilledHours += s.BilledHours
	}
	r.BilledHours = round2(r.BilledHours)

	sortBuckets(r.Buckets, groupBy) // day/tag: label asc; else billed(or hours) desc, tie label asc
	return r
}
```
